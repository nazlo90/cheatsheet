Наростаючі підсумки & frame клаузи

▶
-- Наростаючий підсумок з frame
SELECT
    OrderDate, Amount,
    SUM(Amount) OVER (
        ORDER BY OrderDate
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS RunningTotal,
    -- ковзне середнє за 3 дні
    AVG(Amount) OVER (
        ORDER BY OrderDate
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS MovingAvg3
FROM Orders;
CTE та підзапити Співбесіда

CTE (Common Table Expression)

▶
-- Кілька CTE у ланцюжку
WITH ActiveCustomers AS (
    SELECT CustomerId, Name
    FROM   Customers
    WHERE  IsActive = 1
),
CustomerRevenue AS (
    SELECT ac.CustomerId, SUM(o.Total) AS Revenue
    FROM   ActiveCustomers ac
    JOIN   Orders o ON o.CustomerId = ac.CustomerId
    GROUP BY ac.CustomerId
)
SELECT * FROM CustomerRevenue WHERE Revenue > 1000;
CTE покращують читабельність і можуть використовуватись кілька разів в одному запиті. У SQL Server вони НЕ матеріалізуються за замовчуванням — оптимізатор може їх вбудовувати.

Рекурсивний CTE

▶
-- Обхід організаційної ієрархії
WITH OrgTree AS (
    -- Anchor: співробітники верхнього рівня
    SELECT EmployeeId, Name, ManagerId, 0 AS Level
    FROM   Employees
    WHERE  ManagerId IS NULL

    UNION ALL

    -- Рекурсія: підлеглі
    SELECT e.EmployeeId, e.Name, e.ManagerId, ot.Level + 1
    FROM   Employees e
    JOIN   OrgTree ot ON ot.EmployeeId = e.ManagerId
)
SELECT * FROM OrgTree ORDER BY Level, Name;

    @CustomerId INT,

Статистика & кеш планів

▶
-- Оновити статистику таблиці
UPDATE STATISTICS dbo.Orders;

-- Очистити кеш планів (лише для розробки/тестування!)
DBCC FREEPROCCACHE;

-- Знайти дорогі запити в кеші планів
SELECT TOP 10
    qs.total_worker_time / qs.execution_count AS AvgCPU,
    SUBSTRING(qt.text, (qs.statement_start_offset/2)+1,
        ((qs.statement_end_offset - qs.statement_start_offset)/2)+1) AS QueryText
FROM   sys.dm_exec_query_stats qs
CROSS APPLY sys.dm_exec_sql_text(qs.sql_handle) qt
ORDER BY AvgCPU DESC;

Тригери (T-SQL)

▶
CREATE OR ALTER TRIGGER trg_Orders_AuditUpdate
ON dbo.Orders
AFTER UPDATE
AS
BEGIN
    SET NOCOUNT ON;
    INSERT INTO OrderAudit (OrderId, OldTotal, NewTotal, ChangedAt)
    SELECT d.OrderId, d.Total, i.Total, GETDATE()
    FROM   deleted d
    JOIN   inserted i ON i.OrderId = d.OrderId
    WHERE  d.Total <> i.Total;
END;
Тригери виконуються в тій самій транзакції, що й DML-оператор. Важка логіка всередині тригерів може спричинити конкуренцію за блокування та сповільнити операції запису.

Аномалії та рівні ізоляції

▶

Рівень ізоляції транзакцій визначає, які аномалії можливі при паралельній роботі транзакцій. Чим вищий рівень — тим безпечніше, але повільніше.

Lost UpdateДві транзакції читають, змінюють і записують — одне оновлення губиться

Деталі кожного рівня + приклади

▶
-- READ UNCOMMITTED — бачить незакомічені зміни (dirty reads)
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;

-- READ COMMITTED — дефолт у більшості СУБД
-- Два однакових SELECT у одній транзакції можуть дати різний результат
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
-- T1: SELECT balance FROM accounts WHERE id=1  → 1000
-- T2: UPDATE accounts SET balance=500 WHERE id=1; COMMIT;
-- T1: SELECT balance FROM accounts WHERE id=1  → 500 (non-repeatable read!)

-- REPEATABLE READ — рядки прочитані у транзакції не змінюються
-- Але нові INSERT від іншої транзакції можуть зʼявитись — phantom reads
-- Примітка: PostgreSQL вирішує phantom reads через MVCC навіть на цьому рівні
SET TRANSACTION ISOLATION LEVEL REPEATABLE READ;

-- SERIALIZABLE — повний захист; транзакції виглядають як послідовні
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;

-- PostgreSQL snapshot isolation
BEGIN;
SET TRANSACTION ISOLATION LEVEL REPEATABLE READ;
SELECT * FROM orders WHERE user_id = 42;
-- бачимо snapshot на момент початку транзакції
COMMIT;

MVCC & практичні поради

▶

MVCC (Multi-Version Concurrency Control) — PostgreSQL і MySQL InnoDB зберігають декілька версій рядків замість блокувань читання. Читачі не блокують записників і навпаки.

СценарійРекомендований рівеньПричина
Звіти / аналітика (read-only)REPEATABLE READ або SERIALIZABLEКонсистентний snapshot на весь запит
OLTP (замовлення, платежі)READ COMMITTED + явні блокуванняБаланс між безпекою та продуктивністю
Фінансові операціїSERIALIZABLEЗахист від lost update та phantom reads
Лічильники / чергиREAD COMMITTED + SELECT FOR UPDATEОптимістичне або песимістичне блокування
-- Pessimistic lock — блокує рядок для оновлення
BEGIN;
SELECT balance FROM accounts WHERE id = 1 FOR UPDATE;
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
COMMIT;

-- Optimistic lock — перевірка версії перед оновленням
UPDATE accounts
SET balance = 900, version = version + 1
WHERE id = 1 AND version = 5;
-- якщо 0 рядків оновлено — конфлікт, повторити транзакцію
Для більшості OLTP-задач достатньо READ COMMITTED + SELECT FOR UPDATE на критичних ресурсах. SERIALIZABLE застосовуйте точково — він може значно знизити throughput під навантаженням.
