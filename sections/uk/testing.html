describe('UserCardComponent', () => {
  let fixture: ComponentFixture<UserCardComponent>;
  let svc: jasmine.SpyObj<UsersService>;

  beforeEach(async () => {
    svc = jasmine.createSpyObj('UsersService', ['getUser']);
    await TestBed.configureTestingModule({
      imports: [UserCardComponent],
      providers: [{ provide: UsersService, useValue: svc }]
    }).compileComponents();
    fixture = TestBed.createComponent(UserCardComponent);
  });

  it('displays user name', () => {
    fixture.componentInstance.user = { name: 'Alice' };
    fixture.detectChanges();
    expect(fixture.nativeElement.querySelector('.name').textContent).toContain('Alice');
  });
});

Асинхронне тестування — fakeAsync і RxJS Marbles Ключове

fakeAsync / tick — керування часом у тестах

▶

fakeAsync виконує тіло тесту у фіктивній асинхронній зоні — tick(ms) просуває віртуальний час, дозволяючи тестувати setTimeout, debounceTime і Promise синхронно.

// fakeAsync + tick — тест відкладеного пошуку
it('debounces search input', fakeAsync(() => {
  const results: string[][] = [];
  component.searchResults$.subscribe(r => results.push(r));

  component.searchControl.setValue('ng');
  tick(200);                 // недостатньо — debounceTime(300)
  expect(results.length).toBe(0);

  tick(100);                 // разом 300мс — спрацьовує зараз
  expect(results.length).toBe(1);
  discardPeriodicTasks();   // очистити незавершені інтервали
}));

// flushMicrotasks — для Promise / async/await
it('resolves promise', fakeAsync(() => {
  let result = '';
  Promise.resolve('hello').then(v => result = v);
  flushMicrotasks();
  expect(result).toBe('hello');
}));

// waitForAsync — для справжнього async (HttpClient, compileComponents)
it('loads component', waitForAsync(() => {
  fixture.detectChanges();
  fixture.whenStable().then(() => {
    expect(fixture.nativeElement.textContent).toContain('Alice');
  });
}));

RxJS Marble-тестування — візуалізація потоків

▶

Marble-синтаксис описує часові лінії observable у вигляді ASCII-рядків — ідеально для тестування складних ланцюжків операторів без реальних таймерів.

// Marble-синтаксис: - = 10мс кадр, | = завершення, # = помилка, літери = значення
// 'a--b--c|'  означає: емітувати a, чекати 20мс, емітувати b, чекати 20мс, емітувати c, завершити

import { TestScheduler } from 'rxjs/testing';

describe('switchMap operator', () => {
  let scheduler: TestScheduler;

  beforeEach(() => {
    scheduler = new TestScheduler((actual, expected) => {
      expect(actual).toEqual(expected);
    });
  });

  it('cancels previous inner observable', () => {
    scheduler.run(({ cold, hot, expectObservable }) => {
      const source$  = hot('a--b------');            // користувач вводить 'a', потім 'b'
      const innerA$  = cold('--x|');                  // відповідь для 'a'
      const innerB$  = cold('--y|');                  // відповідь для 'b'
      const expected = '-----y----';                  // 'a' скасовано, тільки результат 'b'

      const result$ = source$.pipe(
        switchMap(v => v === 'a' ? innerA$ : innerB$)
      );

      expectObservable(result$).toBe(expected, { y: 'result-b' });
    });
  });
});
СимволЗначення
-Один віртуальний кадр часу (10мс)
a–zЕмітоване значення (зіставляється через об'єкт values)
|Завершення
#Помилка
^Точка підписки (гарячі observable)
!Точка відписки
Використовуйте cold() для HTTP-подібних observable (починаються заново на кожного підписника). Використовуйте hot() для спільних потоків, як-от події користувача. Marbles роблять складні помилки з часуванням одразу видимими.

Основи та Page Object Model

▶
test('login flow', async ({ page }) => {
  await page.goto('/login');
  await page.getByTestId('email').fill('user@example.com');
  await page.getByRole('button', { name: 'Login' }).click();
  await expect(page).toHaveURL(/dashboard/);
});

// Мок API
await page.route('**/api/users', route => route.fulfill({ json: [] }));

// Page Object Model — завжди згадуйте!
class LoginPage {
  constructor(private page: Page) {}
  async login(email: string, pass: string) {
    await this.page.getByTestId('email').fill(email);
    await this.page.getByTestId('pwd').fill(pass);
    await this.page.getByRole('button', { name: 'Login' }).click();
  }
}
Використовуйте атрибути data-testid — стабільні, не прив'язані до CSS. POM = повторно використовувані, підтримувані селектори. E2E: покривайте лише критичні сценарії користувача.
