Memory Leak Prevention Must Know

3 patterns to prevent subscription leaks

▶

Every subscribe() that isn't cleaned up leaks memory — the callback keeps a reference to the component even after it's destroyed.

// ① async pipe — PREFERRED for templates (auto-unsubscribes)
@Component({ template: `<li *ngFor="let u of users$ | async">{{ u.name }}</li>` })
class UserListComponent {
  users$ = this.http.get<User[]>('/api/users');
  // No subscribe(), no cleanup needed
}

// ② takeUntilDestroyed (Angular 16+) — PREFERRED in class body
class UserListComponent {
  private destroyRef = inject(DestroyRef);

  ngOnInit() {
    interval(1000).pipe(
      takeUntilDestroyed(this.destroyRef)
    ).subscribe(tick => this.update(tick));
    // completes automatically when component is destroyed
  }
}

// ③ Manual unsubscribe — fallback for complex cases
class UserListComponent implements OnDestroy {
  private subs = new Subscription();

  ngOnInit() {
    this.subs.add(stream1$.subscribe(...));
    this.subs.add(stream2$.subscribe(...));
  }
  ngOnDestroy() { this.subs.unsubscribe(); }  // cleans all at once
}

// Bonus — toSignal() also auto-unsubscribes
users = toSignal(this.http.get<User[]>('/api/users'), { initialValue: [] });
// No subscribe, no cleanup — Angular manages it internally

shareReplay — refCount trap

▶

shareReplay multicasts and replays the last N values. The refCount option controls whether the source subscription is kept alive when subscriber count drops to zero.

// ⚠ MEMORY LEAK — refCount: false (the default before RxJS 6.6)
users$ = this.http.get('/api/users').pipe(
  shareReplay({ bufferSize: 1, refCount: false })
);
// Even after all subscribers unsubscribe, the inner subscription STAYS OPEN
// If provided in 'root', leaks for the life of the app — acceptable
// If in a component destroyed and recreated — new source subscription each time!

// ✓ SAFE — refCount: true (recommended)
users$ = this.http.get('/api/users').pipe(
  shareReplay({ bufferSize: 1, refCount: true })
);
// Unsubscribes from source when last subscriber leaves
// BUT: next subscriber triggers a new HTTP request (no true caching)

// Rule: HTTP in a root service → refCount: false is fine (singleton)
// HTTP in component-level service → refCount: true to avoid leaks
