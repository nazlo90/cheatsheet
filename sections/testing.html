// ─── TestBed — component with template ────────────────────
describe('UserCardComponent', () => {
  let fixture: ComponentFixture<UserCardComponent>;
  let svc: jasmine.SpyObj<UsersService>;

  beforeEach(async () => {
    svc = jasmine.createSpyObj('UsersService', ['getUser']);
    await TestBed.configureTestingModule({
      imports: [UserCardComponent],
      providers: [{ provide: UsersService, useValue: svc }]
    }).compileComponents();
    fixture = TestBed.createComponent(UserCardComponent);
  });

  it('displays user name', () => {
    fixture.componentInstance.user = { name: 'Alice' };
    fixture.detectChanges();
    expect(fixture.nativeElement.querySelector('.name').textContent).toContain('Alice');
  });
});

// ─── Isolated — service, no Angular overhead ──────────────
describe('CartService', () => {
  let service: CartService;

  beforeEach(() => {
    service = new CartService();  // plain instantiation
  });

  it('calculates total', () => {
    service.add({ price: 10, qty: 2 });
    expect(service.total()).toBe(20);
  });
});
  • Testing pyramid: Many unit → fewer integration → few E2E
  • AAA: Arrange → Act → Assert
  • Test behavior, not implementation
  • Use HttpClientTestingModule + HttpTestingController for HTTP
Async Testing — fakeAsync & RxJS Marbles Testing

fakeAsync / tick — control time in tests

▶

fakeAsync runs the test body in a fake async zone — tick(ms) advances virtual time, letting you test setTimeout, debounceTime, and Promises synchronously.

// fakeAsync + tick — test debounced search
it('debounces search input', fakeAsync(() => {
  const results: string[][] = [];
  component.searchResults$.subscribe(r => results.push(r));

  component.searchControl.setValue('ng');
  tick(200);                 // not enough — debounceTime(300)
  expect(results.length).toBe(0);

  tick(100);                 // total 300ms — fires now
  expect(results.length).toBe(1);
  discardPeriodicTasks();   // clean up any pending intervals
}));

// flushMicrotasks — for Promises / async/await
it('resolves promise', fakeAsync(() => {
  let result = '';
  Promise.resolve('hello').then(v => result = v);
  flushMicrotasks();
  expect(result).toBe('hello');
}));

// waitForAsync — for real async (HttpClient, compileComponents)
it('loads component', waitForAsync(() => {
  fixture.detectChanges();
  fixture.whenStable().then(() => {
    expect(fixture.nativeElement.textContent).toContain('Alice');
  });
}));

RxJS Marble testing — visualize streams

▶

Marble syntax describes observable timelines as ASCII strings — perfect for testing complex operator chains without real timers.

// Marble syntax: - = 10ms frame, | = complete, # = error, letters = values
// 'a--b--c|'  means: emit a, wait 20ms, emit b, wait 20ms, emit c, complete

import { TestScheduler } from 'rxjs/testing';

describe('switchMap operator', () => {
  let scheduler: TestScheduler;

  beforeEach(() => {
    scheduler = new TestScheduler((actual, expected) => {
      expect(actual).toEqual(expected);
    });
  });

  it('cancels previous inner observable', () => {
    scheduler.run(({ cold, hot, expectObservable }) => {
      const source$  = hot('a--b------');            // user types 'a', then 'b'
      const innerA$  = cold('--x|');                  // response for 'a'
      const innerB$  = cold('--y|');                  // response for 'b'
      const expected = '-----y----';                  // 'a' cancelled, only 'b' result

      const result$ = source$.pipe(
        switchMap(v => v === 'a' ? innerA$ : innerB$)
      );

      expectObservable(result$).toBe(expected, { y: 'result-b' });
    });
  });
});
SymbolMeaning
-One virtual time frame (10ms)
a–zEmitted value (mapped via values object)
|Complete
#Error
^Subscription point (hot observables)
!Unsubscription point
Use cold() for HTTP-like observables (start fresh per subscriber). Use hot() for shared streams like user events. Marbles make complex timing bugs immediately visible.

Playwright E2E Testing

Essentials + Page Object Model

▶
test('login flow', async ({ page }) => {
  await page.goto('/login');
  await page.getByTestId('email').fill('user@example.com');
  await page.getByRole('button', { name: 'Login' }).click();
  await expect(page).toHaveURL(/dashboard/);
});

// API mock
await page.route('**/api/users', route => route.fulfill({ json: [] }));

// Page Object Model — always mention!
class LoginPage {
  constructor(private page: Page) {}
  async login(email: string, pass: string) {
    await this.page.getByTestId('email').fill(email);
    await this.page.getByTestId('pwd').fill(pass);
    await this.page.getByRole('button', { name: 'Login' }).click();
  }
}
Use data-testid attributes — stable, not tied to CSS. POM = reusable, maintainable selectors. E2E: cover critical user journeys only.
