TypeScript Basics TS

Annotations, arrays, tuples, enums, unions

▶

TypeScript is a statically typed superset of JavaScript — it adds type annotations and catches type errors at compile time, then compiles to plain JS.

// Annotations
let n: number = 10;
const name: string = "Alice";
let active: boolean = true;
let val: unknown;   // must narrow before use
let x: any;         // escapes type system — avoid
function log(msg: string): void {}

// Arrays & tuples
let arr: number[] = [1, 2, 3];
let tuple: [string, number] = ["age", 30];
let readonly_arr: readonly number[] = [1, 2];

// Enums
enum Role { Admin = 'ADMIN', User = 'USER' }  // string enum
const enum Dir { Up, Down }  // const enum — inlined at compile time

// Union & intersection
type ID = string | number;
type AdminUser = Admin & User;

// Literal types
type Dir = "left" | "right" | "up" | "down";

// Functions
type Fn = (a: number, b: number) => number;
function greet(name: string, greeting?: string): string {
  return `${greeting ?? 'Hello'}, ${name}`;
}

unknown vs any

▶

Both accept any value, but unknown is the type-safe counterpart — you must narrow it before use, while any silently disables all type checking.

unknownany
Accepts any valueYes ✓Yes ✓
Assignable to other typesNo — must narrow firstYes — bypasses checks
Property/method accessError until narrowedAlways allowed
Type safetyPreserved ✓Disabled ✗
Use caseExternal data, API responsesMigration, quick escape hatch
let a: unknown = fetchData();
a.toFixed();            // ✗ Error — not narrowed yet
if (typeof a === 'number') a.toFixed(); // ✓ narrowed

let b: any = fetchData();
b.toFixed();            // ✓ no error — but runtime may explode
b.foo.bar.baz;          // ✓ TS is silent, runtime throws

// Prefer unknown for function params accepting arbitrary input
function parse(val: unknown): string {
  if (typeof val === 'string') return val;
  return String(val);
}
type vs interface TS

Key differences

▶

Both interface and type describe object shapes — but interface supports declaration merging and is preferred for class contracts, while type is more flexible for unions, mapped types, and aliases.

Structural vs Nominal Typing Interview

Duck typing — shape over name

▶

TypeScript uses structural typing: compatibility is based on the shape of a type (its properties), not the name or declaration. Two types with the same structure are assignable to each other even if unrelated.

// Structural typing — shape matches = compatible
interface Point { x: number; y: number; }
class Vector { x = 0; y = 0; }

const p: Point = new Vector();  // ✓ — same shape

// Excess property check — only on object LITERALS
const p2: Point = { x: 1, y: 2, z: 3 };  // ✗ Error — extra prop z
const obj = { x: 1, y: 2, z: 3 };
const p3: Point = obj;               // ✓ — not a literal, no excess check

// Angular service impact — structural typing means two services
// with identical public APIs are interchangeable at the type level:
class RealAuthService  { isLoggedIn() { return true; } }
class MockAuthService  { isLoggedIn() { return false; } }
// TestBed can swap them without a shared interface — structural match is enough

// Nominal typing simulation — use branded types to prevent accidental mixing
type UserId  = string & { readonly _brand: 'UserId'  };
type OrderId = string & { readonly _brand: 'OrderId' };

function asUserId(id: string): UserId { return id as UserId; }

function getUser(id: UserId) { /* ... */ }
const orderId = asOrderId('abc');
getUser(orderId);  // ✗ Error — OrderId is not assignable to UserId
// Without branding: both are string → silently accepted
Structural (TypeScript)Nominal (Java, C#)
Compatibility checkShape of propertiesExplicit class/interface name
Two identical shapesAlways compatible ✓Not compatible without declaration
Testing benefitMock by shape — no shared interface neededMust implement the same interface
RiskAccidental compatibility (UserId ↔ OrderId)Boilerplate, rigid hierarchies
Use branded types when domain correctness matters (IDs, currencies, units). Use structural typing freely for mocks and DTOs — it's a feature, not a bug.

Generics, mapped, conditional, infer

▶
// Generic with constraint
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

// Mapped type
type Nullable<T> = { [K in keyof T]: T[K] | null };
type Mutable<T> = { -readonly [K in keyof T]: T[K] };   // remove readonly
type Optional<T> = { [K in keyof T]?: T[K] };           // add ?

// Conditional + infer
type UnwrapPromise<T> = T extends Promise<infer R> ? R : T;
type IsArray<T> = T extends any[] ? 'yes' : 'no';

// Template literal types
type EventName<T extends string> = `on${Capitalize<T>}`;

// Discriminated union + narrowing
type Shape = { kind: 'circle'; r: number } | { kind: 'rect'; w: number; h: number };
function area(s: Shape): number {
  switch (s.kind) {
    case 'circle': return Math.PI * s.r ** 2;
    case 'rect':   return s.w * s.h;
  }
}

// Deep partial (recursive)
type DeepPartial<T> = T extends object
  ? { [K in keyof T]?: DeepPartial<T[K]> } : T;

Type-safe API client — mapped + conditional types in practice

▶

Combine mapped types, conditional types, and template literals to build an HTTP client where every endpoint's params and response are fully typed at compile time.

// 1. Define the API contract as a type map
interface ApiRoutes {
  '/users':        { method: 'GET';  params: { page: number };   response: User[]   };
  '/users/:id':    { method: 'GET';  params: { id: string };      response: User     };
  '/users/create': { method: 'POST'; params: { body: NewUser };   response: User     };
}

// 2. Extract helpers via conditional types
type GetRoutes  = { [K in keyof ApiRoutes]: ApiRoutes[K]['method'] extends 'GET'  ? K : never }[keyof ApiRoutes];
type PostRoutes = { [K in keyof ApiRoutes]: ApiRoutes[K]['method'] extends 'POST' ? K : never }[keyof ApiRoutes];

type RouteParams<T extends keyof ApiRoutes>   = ApiRoutes[T]['params'];
type RouteResponse<T extends keyof ApiRoutes> = ApiRoutes[T]['response'];

// 3. Typed client function
function apiGet<T extends GetRoutes>(
  url: T,
  params: RouteParams<T>
): Observable<RouteResponse<T>> {
  return http.get<RouteResponse<T>>(url, { params: params as any });
}

// 4. Usage — fully typed, wrong args = compile error
apiGet('/users', { page: 1 });         // ✓ response: User[]
apiGet('/users/:id', { id: 'abc' });  // ✓ response: User
apiGet('/users', { id: 'abc' });      // ✗ Error — wrong params for /users

// Template literal type — generate event names automatically
type Endpoint = keyof ApiRoutes;
type LoadAction<T extends string> = `load${Capitalize<T>}`;
type UserActions = LoadAction<'users' | 'orders'>;  // 'loadUsers' | 'loadOrders'
This pattern is used by tRPC and typed fetch wrappers — the client type is derived entirely from the route map, so adding a new endpoint in one place updates the whole client automatically.
Abstract Classes vs Interfaces TSInterview

Core differences at a glance

▶

Abstract classes can contain implementation (methods, fields, constructors) and enforce a contract — they sit halfway between a concrete class and an interface. Interfaces are pure shape contracts: zero runtime footprint, erased at compile time.

FeatureAbstract classInterface
Runtime existenceYes — compiled to JS classNo — erased at compile time
Concrete methods / fieldsYes ✓No ✗ (only signatures)
Constructor logicYes ✓No ✗
Multiple inheritanceNo — single extendsYes — multiple implements
Declaration mergingNo ✗Yes ✓
Use with instanceofYes ✓No ✗
Dependency Injection tokenYes ✓ (Angular/NestJS)No — not a runtime value
Default rule: prefer interface for pure shape contracts. Reach for abstract class when you need shared implementation, a constructor, or a runtime DI token.

Abstract class — shared implementation + contract

▶
// Abstract class — cannot be instantiated directly
abstract class Animal {
  // Concrete field + constructor — shared by all subclasses
  constructor(protected readonly name: string) {}

  // Concrete method — inherited as-is
  describe(): string {
    return `I am ${this.name}`;
  }

  // Abstract method — subclasses MUST implement
  abstract makeSound(): string;
}

class Dog extends Animal {
  makeSound() { return 'Woof'; }
}

const d = new Dog('Rex');
d.describe();   // 'I am Rex' — inherited from Animal
d.makeSound();  // 'Woof'

// new Animal('x') — ✗ Error: Cannot create instance of abstract class
d instanceof Animal; // ✓ true — runtime check works

Interface — pure shape, zero runtime cost

▶
// Interface — type-level only, compiled away
interface Serializable {
  serialize(): string;
  deserialize(raw: string): void;
}

// A class can implement multiple interfaces
interface Loggable { log(): void; }

class UserRecord implements Serializable, Loggable {
  serialize()          { return JSON.stringify(this); }
  deserialize(raw)     { Object.assign(this, JSON.parse(raw)); }
  log()                { console.log(this); }
}

// Declaration merging — add fields to existing interface
interface Serializable { version: number; }  // ✓ merged

// instanceof with interface — ✗ doesn't work (no runtime token)
// ur instanceof Serializable  — compile error

// Use a type guard instead:
function isSerializable(v: unknown): v is Serializable {
  return typeof (v as any)?.serialize === 'function';
}

// Angular DI — abstract class as token (works at runtime)
abstract class Logger { abstract log(msg: string): void; }
class ConsoleLogger extends Logger { log(m: string) { console.log(m); } }

// providers array in Angular module:
// { provide: Logger, useClass: ConsoleLogger }
// inject(Logger) — ✓ resolves to ConsoleLogger at runtime

// Interface token — ✗ does NOT work
interface ILogger { log(msg: string): void; }
// { provide: ILogger, useClass: ... } — compile error: ILogger not a value
In Angular / NestJS: use abstract classes as DI tokens — they survive compilation, interfaces don't. The abstract class acts as both the type contract and the injection key.
